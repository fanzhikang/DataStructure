# 排序

## 插入排序

~~~c
void Insertion_Sort(ElementType a[], int N)
  {
    int p;i;
    ElementType tmp;
    for(p=1;p<N;p++){
      tmp = a[p];      //摸下一张牌
      for(i=p;i>0&&a[i-1]>tmp;i--)
        a[i]=a[i-1];   //移出空位
      a[i]=tmp;   //新牌落位
    }
  }
~~~

## 希尔排序

~~~c
void Shell_sort(ElementType a[],int n)
  {
    for (d=n/2;d>0;d/=2){   //希尔增量序列
      for (p=d;p<n;p++){    //插入排序
        tmp = a[p];
        for(i=p;i>=d&&a[i-d]>tmp;i-=d)
          a[i]=a[i-d];
        a[i]=tmp;
      }
    }
  }
~~~
